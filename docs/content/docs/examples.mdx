### Localized Pages

<Tabs items={['Next.js', 'TanStack Start']}>
  <Tab value="Next.js">
    ```tsx title="app/[locale]/page.tsx"
    import { metadataClient } from "@/lib/metadata";

    interface LocalizedPageProps {
      params: { locale: string };
    }

    export const generateMetadata = metadataClient.getMetadata(
      ({ params }: LocalizedPageProps) => ({
        path: `/${params.locale}`,
        fallback: {
          title: params.locale === 'es' ? "Inicio - Mi Aplicación" : "Home - My App",
          description: params.locale === 'es'
            ? "Bienvenido a nuestra aplicación increíble"
            : "Welcome to our amazing application",
          openGraph: {
            locale: params.locale === 'es' ? 'es_ES' : 'en_US'
          }
        },
        override: {
          alternates: {
            languages: {
              'en': 'https://myapp.com/en',
              'es': 'https://myapp.com/es'
            }
          }
        }
      })
    );

    export default function LocalizedPage({ params }: LocalizedPageProps) {
      return (
        <div>
          <h1>{params.locale === 'es' ? 'Bienvenido' : 'Welcome'}</h1>
        </div>
      );
    }
    ```

## Development Workflow Examples

### Environment-Based Configuration

```ts title="lib/metadata.ts"
import { GenerateMetadataClient } from "generate-metadata/next";

export const metadataClient = new GenerateMetadataClient({
  dsn:
    process.env.NODE_ENV === "development"
      ? undefined // Skip API calls in development
      : process.env.NODE_ENV === "staging"
        ? process.env.NEXT_PUBLIC_GENERATE_METADATA_DSN_STAGING
        : process.env.NEXT_PUBLIC_GENERATE_METADATA_DSN,
});
```

### A/B Testing Integration

```tsx title="app/landing/page.tsx"
import { metadataClient } from "@/lib/metadata";

export const generateMetadata = metadataClient.getMetadata(async () => {
  // Get A/B test variant
  const variant = await getABTestVariant("landing-metadata");

  return {
    path: `/landing?variant=${variant}`,
    fallback: {
      title:
        variant === "A"
          ? "My App - Best Solution"
          : "My App - Revolutionary Platform",
      description:
        variant === "A"
          ? "The best solution for your business needs"
          : "Revolutionary platform that changes everything",
    },
  };
});
```

### Error Boundary with Fallback

```tsx title="app/error-prone/page.tsx"
import { metadataClient } from "@/lib/metadata";

export const generateMetadata = metadataClient.getMetadata(() => ({
  path: "/error-prone",
  fallback: {
    // Always provide robust fallbacks
    title: "Page Title - My App",
    description: "Page description that always works",
    openGraph: {
      title: "Page Title - My App",
      description: "Page description that always works",
    },
  },
  // If API fails, fallback metadata ensures SEO still works
}));
```
